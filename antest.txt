function Stack () {

	return [
		{"class" : "Stack"},
		{"next"  : 0},
		{"data"  : []},
		{"empty" : (function (self){
			return self.next == 0;
		})},
		
		{"push"	 : (function (self, data){
			self.data[self.next++] = data;
		})},
		
		{"pop"	 : (function (self){
			if (self.next > 0)
				self.data[--self.next] = nil;
		})},
		
		{"top" 	 : (function (self){
			return self.data[self.next-1];
		})}
	];
}
